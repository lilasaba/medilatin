% Challenges block.
\begin{textblock}{7}(0,4)
  \LHead{Challenges}

  \begin{itemize}
      \item Latin is not spoken natively 
      \item There is no available speech database, and it is resource-heavy to create one
      \item Many variants/dialects exists, and we can only make guesses about the pronunciation
      \item The pronunciation mainly depends on
      \begin{itemize}
          \item the \textbf{era} of the read text 
          \item the \textbf{georaphical region} where the text originates from
          \item the \textbf{native language} of the speaker
      \end{itemize}
  \end{itemize}

\end{textblock}

% LM.
\begin{textblock}{3}(0,11.5)
   \LHead{Language model}

\begin{modelbox}
\begin{itemize}
\item 3-gram language model
\item Kneser-Ney smoothing
\item Interpolating the two corpora
\item SRILM~\cite{srilm}
\end{itemize}
\end{modelbox}

\end{textblock}

% AM.
\begin{textblock}{3}(4,11.5)
   \LHead{Acoustic model}

\begin{modelbox}
\begin{itemize}
\item 6-hidden-layer DNN
\item 2000 neurons per layer
\item p-norm activation function
\item 7000-11000 senones (softmax size)
\item Kaldi toolkit~\cite{kaldi}
\end{itemize}
\end{modelbox}

\end{textblock}

\LHead{Conclusions}

\begin{itemize}

\item Four-language USG modeling is the best
\begin{itemize}
\item It is able to generalize over different speaker test sets
\end{itemize}
\end{itemize}
